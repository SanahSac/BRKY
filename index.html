function getPhaseInfo(){

  const start=new Date(document.getElementById('periodStart').value);

  const cl=parseInt(document.getElementById('cycleLen').value)||32;

  const pl=parseInt(document.getElementById('periodLen').value)||3;

  const today=new Date();

  const raw=Math.floor((today-start)/86400000)+1;

  const d=((raw-1)%cl)+1;

  const ov=Math.round(cl/2)-2;

  let phase;

  if(d<=pl)phase='menstrual';

  else if(d<=ov-1)phase='follicular';

  else if(d<=ov+2)phase='ovulatory';

  else phase='luteal';

  return{phase,day:d,cl,daysLeft:cl-d+1,pl,ov};

}

function fmtAmt(a,s){

  const v=a*s;

  if(v===Math.round(v))return String(v);

  const fr={0.25:'¼',0.5:'½',0.75:'¾',0.33:'⅓',0.67:'⅔'};

  const w=Math.floor(v),f=parseFloat((v-w).toFixed(2));

  const fs=fr[f]||v.toFixed(1);

  return w>0?`${w} ${fs}`:fs;

}

      <div><h4 style="color:${c.text}">Best fruits this phase</h4><p style="color:${c.text}">${s.fruits}</p><p class="why" style="color:${c.text}">${s.fruitWhy}</p></div>

      <label>Servings</label>

      <button class="srv-btn" onclick="chgSrv(-1)">−</button>

    </div>

    <div class="rsec"><h3>Ingredients</h3><ul id="ingList"></ul></div>

    <div class="rsec"><h3>Steps</h3><ol>${m.steps.map(st=>`<li>${st}</li>`).join('')}</ol></div>

    ${m.nightPrep?`<div class="rsec"><h3>${isBolton?'Bolton notes':'Night-before prep'}</h3><p style="font-size:13px;color:var(--text)">${m.nightPrep}</p></div>`:''}

  </div>`;

}


function updMacros(){

  if(selDay===null)return;

  const m=getWeekMeals()[selDay];

  document.getElementById('macros').innerHTML=`

    <div class="mc"><span class="mc-val">${Math.round(m.protein*servings)}g</span><div class="mc-lbl">Protein</div></div>

    <div class="mc"><span class="mc-val">${Math.round(m.carbs*servings)}g</span><div class="mc-lbl">Carbs</div></div>

    <div class="mc"><span class="mc-val">${Math.round(m.fibre*servings)}g</span><div class="mc-lbl">Fibre</div></div>`;

  document.getElementById('srvCount').textContent=servings;

  document.getElementById('ingList').innerHTML=m.ingredients.map(ing=>`<li>${fmtAmt(ing.a,servings)}${ing.u?' '+ing.u:''} ${ing.n}</li>`).join('');

}


function chgSrv(d){servings=Math.max(1,Math.min(6,servings+d));updMacros();}

function addIngredient(){

  const inp=document.getElementById('ingInput');

  const val=inp.value.trim().toLowerCase();

  if(!val)return;

  val.split(',').map(v=>v.trim()).filter(Boolean).forEach(p=>{if(p&&!myIngredients.includes(p))myIngredients.push(p);});

  inp.value='';

  renderMatcher();

}

function removeIngredient(i){myIngredients.splice(i,1);renderMatcher();}


async function renderMatcher(){

  document.getElementById('tagCloud').innerHTML=myIngredients.map((ing,i)=>

    `<div class="ing-tag">${ing}<span class="rm" onclick="removeIngredient(${i})">×</span></div>`

  ).join('');

  const results=document.getElementById('matchResults');

  if(!myIngredients.length){results.innerHTML='<p class="empty">Add ingredients above to get AI suggestions</p>';return;}

  if(aiLoading)return;

  aiLoading=true;

  results.innerHTML='<p class="loading">Finding the best breakfasts for you...</p>';

  const info=getPhaseInfo();

  const phaseLabel=PC[info.phase].label;

  const c=PC[info.phase];

  const prompt=`You are BRKY, a cycle-aware vegetarian breakfast advisor.

The user is in their ${phaseLabel} phase. They are vegetarian and do NOT eat eggs (their husband does).

They have: ${myIngredients.join(', ')}.

They also likely have: olive oil, salt, pepper, basic spices (cumin, turmeric, coriander), water, garlic, onion, ginger, vinegar, lemon.

Suggest exactly 4 warm vegetarian breakfasts. Vary cuisines globally — Japanese, Mexican, Mediterranean, Korean, Middle Eastern, Indian, etc. Make them protein-rich and warming. Max 6 ingredients and 20 min each. At least one should be relevant to the ${phaseLabel} phase.

Respond ONLY in this exact JSON (no markdown, no extra text):

[{"name":"","cuisine":"","desc":"One sentence.","protein":0,"carbs":0,"fibre":0,"time":"","phaseNote":"Why good for ${phaseLabel} phase, one sentence.","steps":["","","",""],"extraNeeded":[]},...]`;


  try{

    const res=await fetch('https://api.anthropic.com/v1/messages',{

      method:'POST',

      headers:{'Content-Type':'application/json'},

      body:JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:1200,messages:[{role:'user',content:prompt}]})

    });

    const data=await res.json();

    const txt=data.content.filter(b=>b.type==='text').map(b=>b.text).join('');

    const suggestions=JSON.parse(txt.replace(/```json|```/g,'').trim());

    results.innerHTML=suggestions.map(s=>`<div class="match-card">

      <h3>${s.name} <span class="ai-tag">AI</span></h3>

      <div class="match-meta">${s.cuisine} · ${s.time}</div>

      <p class="match-desc">${s.desc}</p>

      <div style="display:flex;gap:5px;flex-wrap:wrap;margin-bottom:6px">

        <span class="badge bp">${s.protein}g protein</span>

        <span class="badge bt">${s.carbs}g carbs</span>

        <span class="badge bt">${s.fibre}g fibre</span>

      </div>

      <div style="font-size:12px;padding:6px 9px;border-radius:var(--radius-sm);background:${c.bg};color:${c.text};margin-bottom:8px">${s.phaseNote}</div>

      <div class="match-steps rsec"><h3>Steps</h3><ol>${s.steps.map(st=>`<li>${st}</li>`).join('')}</ol></div>

      ${s.extraNeeded&&s.extraNeeded.length?`<p class="match-need">Might also need: ${s.extraNeeded.join(', ')}</p>`:''}

    </div>`).join('');

  }catch(e){

    results.innerHTML='<p class="empty">Could not get suggestions — check your connection and try again.</p>';

  }

  aiLoading=false;

}
